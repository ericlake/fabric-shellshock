Running commands on a server

Lets find out what version of RedHat is running on a remote server

1 from fabric.api import *
2 
3 @task
4 def get_distro_version():
5     "Report the distro version of the remote|local system"
6     result = run('cat /etc/redhat-release')
7     puts(result)
